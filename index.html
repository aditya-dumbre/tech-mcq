🔓 Employee System — Solution Skeleton
import mysql.connector
import numpy as np
import matplotlib.pyplot as plt
from abc import ABC, abstractmethod

class EmployeeNotFoundError(Exception): pass
class SalaryValidationError(Exception): pass

class StaffMember(ABC):
    @abstractmethod
    def role(self): pass

class Employee(StaffMember):
    def __init__(self, emp_id, name, salary):
        self.emp_id = emp_id; self.name = name; self.__salary = salary
    @property
    def salary(self): return self.__salary
    @salary.setter
    def salary(self, v):
        if v < 0: raise SalaryValidationError(f"Invalid: {v}")
        self.__salary = v
    def role(self): return "Employee"

class Manager(Employee):
    def __init__(self, emp_id, name, salary, team=[]):
        super().__init__(emp_id, name, salary); self.team = team
    def role(self): return "Manager"

class Department:  # Composition — owns employees
    def __init__(self, name):
        self.name = name; self.employees = []
    def add(self, emp): self.employees.append(emp)

class Company:  # Aggregation — references depts
    def __init__(self, depts): self.depts = depts
    def all_salaries(self):
        return np.array([e.salary for d in self.depts for e in d.employees])

def add_employee(name, dept_id, salary):
    conn = mysql.connector.connect(host="localhost",user="root",password="",database="empdb")
    try:
        cur = conn.cursor()
        cur.execute("INSERT INTO employees(name,dept_id,salary) VALUES(%s,%s,%s)",(name,dept_id,salary))
        conn.commit()
    except mysql.connector.Error as e:
        raise EmployeeNotFoundError("Insert failed") from e
    finally: conn.close()

🔓 Grade Tracker — Solution Skeleton
import mysql.connector
import numpy as np
import pandas as pd

class InvalidGradeError(Exception): pass
class StudentNotFoundError(Exception): pass

class Mark:  # Association class
    def __init__(self, student_id, subject_id, score):
        if not 0 <= score <= 100: raise InvalidGradeError(f"Bad: {score}")
        self.student_id = student_id; self.subject_id = subject_id; self.score = score

class Student:
    def __init__(self, sid, name):
        self.sid = sid; self.name = name; self.__marks = []
    def add_mark(self, m): self.__marks.append(m)
    @property
    def gpa(self):
        return np.mean([m.score for m in self.__marks]) if self.__marks else 0
    @classmethod
    def letter_grade(cls, score):
        return "A" if score>=90 else "B" if score>=75 else "C" if score>=60 else "D" if score>=40 else "F"

class GradeBook:  # Aggregation
    def __init__(self): self.students = []
    def add(self, s): self.students.append(s)
    def rank(self):
        gpas = np.array([s.gpa for s in self.students])
        return [self.students[i] for i in np.argsort(gpas)[::-1]]

def load_marks():
    conn = mysql.connector.connect(host="localhost",user="root",password="",database="gradedb")
    try:
        return pd.read_sql("SELECT s.name student, sub.name subject, m.score FROM marks m JOIN students s ON m.student_id=s.id JOIN subjects sub ON m.subject_id=sub.id", conn)
    finally: conn.close()

🔓 Library System — Solution Skeleton
import mysql.connector
import numpy as np
import pandas as pd
from abc import ABC, abstractmethod

class BookNotAvailableError(Exception): pass
class BookNotFoundError(Exception): pass

class LibraryItem(ABC):
    @abstractmethod
    def availability_status(self): pass

class Book(LibraryItem):
    def __init__(self, isbn, title, available, total):
        self.isbn=isbn; self.title=title; self.available=available; self.total=total
    def availability_status(self): return f"{self.available}/{self.total} copies"

class Catalog:
    def __init__(self): self.books = {}
    def add(self, b): self.books[b.isbn] = b
    def get(self, isbn):
        if isbn not in self.books: raise BookNotFoundError(isbn)
        return self.books[isbn]

class Library:  # Composition
    def __init__(self): self.catalog = Catalog()

def borrow_book(conn, member_id, isbn):
    conn.autocommit = False
    try:
        cur = conn.cursor(dictionary=True)
        cur.execute("SELECT available FROM books WHERE isbn=%s",(isbn,))
        row = cur.fetchone()
        if not row: raise BookNotFoundError(isbn)
        if row["available"]==0: raise BookNotAvailableError()
        cur.execute("UPDATE books SET available=available-1 WHERE isbn=%s",(isbn,))
        cur.execute("INSERT INTO borrow_history(member_id,isbn,borrowed_on) VALUES(%s,%s,NOW())",(member_id,isbn))
        conn.commit()
    except:
        conn.rollback(); raise
    finally: conn.close()

OOP · EncapsulationSalary Guard
class Employee:
    def __init__(self, name, salary):
        self.name = name
        self.__salary = salary  # private

    @property
    def salary(self):
        return self.__salary

    @salary.setter
    def salary(self, val):
        if val < 0:
            raise ValueError("Negative!")
        self.__salary = val

e = Employee("Alice", 50000)
e.salary = 60000      # OK
print(e.salary)        # 60000
# e.__salary → AttributeError
⚙️ Encapsulation — Flow
1
Name Mangling
__salary is renamed to _Employee__salary at compile time. Accessing e.__salary externally fails because that name doesn't exist in the instance dict.
e.__dict__: {'name':'Alice', '_Employee__salary':60000}
2
@property Descriptor
Python stores salary as a descriptor on the class (not the instance). e.salary triggers Employee.salary.__get__(e) — a transparent function call that looks like attribute access.
Employee.__dict__['salary']: property object (has fget, fset)
e.salary: → calls fget(e) → returns _Employee__salary
3
Setter Validation Gate
Assignment e.salary = val triggers fset(e, val). The setter runs validation before touching the private attribute — the caller can never bypass this gate.
e.salary = -1: fset(-1) → val < 0 → raise ValueError
Private attr unchanged — guard held
OOP · InheritanceStaff Hierarchy
class Staff:
    def __init__(self, name, salary):
        self.name = name
        self.salary = salary
    def info(self):
        return f"{self.name} — {self.salary}"

class Manager(Staff):
    def __init__(self, name, salary, dept):
        super().__init__(name, salary)
        self.dept = dept
    def info(self):
        base = super().info()
        return f"{base} [{self.dept}]"

m = Manager("Bob", 90000, "IT")
print(m.info())
# Bob — 90000 [IT]
⚙️ Inheritance — Flow
1
MRO — C3 Linearisation
Python builds a Method Resolution Order for every class. For Manager: [Manager → Staff → object]. When Python looks up a method it walks this list and stops at the first match.
Manager.mro(): [Manager, Staff, object]
2
super().__init__ — Shared Instance
super().__init__(name, salary) runs Staff.__init__ on the same Manager instance (self). Attributes are set on the one object — not on a separate Staff object.
m.__dict__: {name, salary, dept} — all on one object
No separate Staff object exists
3
super().info() — Delegate Up
super().info() inside Manager calls Staff.info(self) using the Manager instance. self still refers to the Manager — so self.name and self.salary are read from the Manager's dict.
Staff.info(self) → self.name = "Bob", self.salary = 90000

# Student & Subject are INDEPENDENT
class Student:
    def __init__(self, sid, name):
        self.sid = sid
        self.name = name

class Subject:
    def __init__(self, subid, name):
        self.subid = subid
        self.name = name

# Association: Mark links them by ID
class Mark:
    def __init__(self, s_id, sub_id, score):
        self.s_id = s_id
        self.sub_id = sub_id
        self.score = score

s1 = Student(1, "Alice")
m1 = Subject(10, "Math")
# Mark exists independently
mk = Mark(s1.sid, m1.subid, 88)
⚙️ Association — Flow
1
Independent Lifetimes
Student, Subject, and Mark are three separate objects on the heap. Deleting s1 doesn't affect mk or m1. This is the key distinction from Composition where child lifetime depends on parent.
s1 refcount → 0: Student object GC'd
mk still alive: holds only the int sid=1, no reference to s1
2
Association via ID (Loose Coupling)
Mark stores s_id and sub_id as plain integers — not object references. This mirrors how a database FK works. Resolving "who is student 1" requires a lookup, not a pointer follow.
mk.__dict__: {s_id:1, sub_id:10, score:88}
No pointer to s1 or m1 objects
3
Many-to-Many via Association Class
One student can have many Mark records. One subject can appear in many Mark records. The association class (Mark) is the junction — exactly how a relational database models many-to-many relationships.
s1 ←→ [mk1, mk2, mk3] (Alice's marks)
m1 ←→ [mk1, mk4, mk7] (Math marks)
OOP · AggregationGradeBook
class Student:
    def __init__(self, name): self.name = name

# Aggregation: GradeBook HOLDS but
# does NOT OWN Students
class GradeBook:
    def __init__(self):
        self.students = []  # reference list

    def enrol(self, student):
        self.students.append(student)

s1 = Student("Alice")
s2 = Student("Bob")

gb = GradeBook()
gb.enrol(s1); gb.enrol(s2)

# s1 exists independently of gb
del gb  # gb gone
print(s1.name)  # Alice — still alive
⚙️ Aggregation — Flow
1
Shared Ownership — Refcounting
When gb.enrol(s1) runs, Python stores a reference to the same Student object that s1 points to. Two variables now point to one object — the refcount goes to 2.
s1 (stack): ref → Student@0x7f10 (refcount=2)
gb.students[0]: ref → same Student@0x7f10
2
del gb — Only GradeBook Dies
del gb removes the GradeBook binding. Python drops the list inside gb, which releases one reference to each Student. But s1 still holds a reference — refcount drops to 1, object survives.
After del gb: Student@0x7f10 refcount = 1 (just s1)
GradeBook object: refcount = 0 → deallocated
3
Contrast: Composition would GC Children
If GradeBook composed students (created them internally, no external reference), del gb would leave refcount = 0 for each student — they'd all be garbage collected immediately.
Composition: del parent → children refcount=0 → GC
Aggregation: del container → children survive via external refs
OOP · CompositionLibrary Owns
class Catalog:
    def __init__(self): self.books = {}
    def add(self, isbn, title):
        self.books[isbn] = title

class BorrowDesk:
    def __init__(self): self.records = []
    def borrow(self, isbn, member):
        self.records.append((isbn, member))

# Composition: Library CREATES and OWNS
class Library:
    def __init__(self):
        self.catalog = Catalog()   # created here
        self.desk = BorrowDesk() # owned by Library

lib = Library()
lib.catalog.add("978-1", "Python")
lib.desk.borrow("978-1", "Alice")
# del lib → catalog & desk also gone
⚙️ Composition — Flow
1
Library Creates Its Parts
Inside Library.__init__, Catalog() and BorrowDesk() are instantiated. Only the Library holds references to them — no external variable ever points to these objects.
Catalog@0x7f20: refcount = 1 (only lib.catalog)
BorrowDesk@0x7f30: refcount = 1 (only lib.desk)
2
Cascading Deletion
When del lib runs, the Library object is GC'd. Its __dict__ (containing catalog and desk) is released. Each child's refcount drops to 0 — they're immediately GC'd too. No survivors.
del lib: Library refcount=0 → GC
→ lib.catalog released → Catalog refcount=0 → GC
→ lib.desk released → BorrowDesk refcount=0 → GC
3
Strong Ownership Semantics
Composition expresses "whole owns part". It enforces tight coupling intentionally — Catalog has no meaning without a Library. This maps directly to the real world: a library catalog doesn't exist independently of a library.
Aggregation: "has-a, shared" — parts survive parent
Composition: "has-a, exclusive" — parts die with parent
OOP · PolymorphismUniform Interface
from abc import ABC, abstractmethod

class LibraryItem(ABC):
    @abstractmethod
    def availability_status(self): pass

class Book(LibraryItem):
    def __init__(self, title, avail, total):
        self.title=title; self.avail=avail
        self.total=total
    def availability_status(self):
        return f"{self.avail}/{self.total} copies"

class Magazine(LibraryItem):
    def __init__(self, title, issue):
        self.title=title; self.issue=issue
    def availability_status(self):
        return f"Issue {self.issue} — on shelf"

items = [Book("Python",3,5), Magazine("IEEE",42)]
for item in items:
    print(item.availability_status())
⚙️ Polymorphism — Flow
1
Duck Typing — No isinstance Needed
Python doesn't check the type of item before calling availability_status(). It just looks up the method on whatever object is there. If the method exists, it runs. This is duck typing.
item = Book: calls Book.availability_status
item = Magazine: calls Magazine.availability_status
— zero type-checking code written
2
Abstract Method — Contract Enforcer
@abstractmethod makes LibraryItem un-instantiable. Python records abstract methods in LibraryItem.__abstractmethods__. Subclasses that don't implement them also become abstract — caught at instantiation time, not runtime.
LibraryItem().__abstractmethods__: {'availability_status'}
Book().__abstractmethods__: set() — all implemented ✓
3
Method Dispatch — vtable in CPython
CPython resolves item.availability_status by looking in the object's type's __dict__, then walking the MRO. This lookup is cached in the type's method cache after the first call — subsequent calls are O(1).
Dispatch: type(item).__dict__ → MRO walk
After first call: cached in type's version tag
OOP · AbstractionHide Complexity
from abc import ABC, abstractmethod
import mysql.connector

class DataStore(ABC):
    @abstractmethod
    def save(self, record): pass
    @abstractmethod
    def fetch(self, key): pass

class MySQLStore(DataStore):
    def __init__(self, conn):
        self.conn = conn

    def save(self, record):
        cur = self.conn.cursor()
        cur.execute("INSERT INTO emp VALUES(%s,%s)",
                   (record["id"], record["name"]))
        self.conn.commit()

    def fetch(self, key):
        cur = self.conn.cursor(dictionary=True)
        cur.execute("SELECT * FROM emp WHERE id=%s",(key,))
        return cur.fetchone()

# Caller uses DataStore interface only
def register(store: DataStore, record):
    store.save(record)  # no MySQL knowledge
⚙️ Abstraction — Flow
1
Abstract Class = Contract Only
DataStore defines what must exist (save, fetch) but says nothing about how. It's a pure interface contract. The caller only needs to know the contract — the implementation is hidden behind it.
DataStore: knows save() and fetch() exist
Caller: never imports mysql.connector — abstraction achieved
2
Swappable Implementation
You can swap MySQLStore for an InMemoryStore or PostgreSQLStore without changing any calling code. register(store, record) works identically regardless of which concrete store is passed.
register(MySQLStore(conn), rec): writes to MySQL
register(InMemoryStore(), rec): writes to dict
— same call, different behaviour
3
Type Hints as Documentation
store: DataStore in the function signature is a type hint — Python doesn't enforce it at runtime, but IDEs, mypy, and code readers know exactly what interface store must conform to. Abstraction + documentation in one.
At runtime: hint ignored — duck typing applies
mypy: checks MySQLStore implements all abstract methods
